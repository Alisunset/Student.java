public class StudentList { private Student[] list = new Student[100]; private int p = 0;

public void add(Student s) {
    list[p++] = s;
}

public Student get(int n) {
    return list[n];
}

public int find(String name, String idintif) {
    if (idintif.equalsIgnoreCase("name")) {
        for (int i = 0; i < p; i++) {
            if (list[i].getName().equalsIgnoreCase(name))
                return i;
        }
    } else if (idintif.equalsIgnoreCase("surname")) {
        for (int i = 0; i < p; i++) {
            if (list[i].getSurname().equalsIgnoreCase(name))
                return i;
        }
    }
    return -1;
}

public int find(Date birth){
    for (int i = 0; i < p; i++){
        if (list[i].getBirth().equals(birth))
            return i;
    }
    return -1;
}

public void delete(int n) {
    if (n >= 0 && n < p) {
            for (int i = n; i < p; i++) {
                list[i] = list[++i];
                i--;
            }p--;
        }
    else return;

}

public void showAll(){
    for(int i = 0; i < p; i++)
        list[i].show();
}
